# TEMPUS TRAINER

Complete training workflow for TEMPUS protocol - teaches AI to understand reality, not chronology.

## Core Architecture

**5 Phases (AP→PR→SA→VE→EC):**

1. **AP (Ancrage Présent):** Establish T₀ anchor point
2. **PR (Positionnement Rétroactif):** Jump to random T₁ in history
3. **SA (Simulation Aveugle):** AI predicts from T₁ (blind to future)
4. **VE (Validation Empirique):** Compare prediction vs reality
5. **EC (Extraction Causale):** Identify causal patterns, calibrate model

**Loop:** Repeat PR→SA→VE→EC for all T₁ points in dataset (spatial sampling)

---

## TRAINING WORKFLOW

**Setup → Train → Calibrate → Validate**

1. **Setup** — Load dataset, initialize model
2. **Train** — Execute TEMPUS loop
3. **Calibrate** — Adjust model based on gaps
4. **Validate** — Test causal understanding

def load_tempus_dataset(dataset_path):
    """
    Load dataset generated by tempus-dataset-generator
    """
    
    # Load data
    with open(dataset_path, 'r') as f:
        dataset = [json.loads(line) for line in f]
    
    # Load metadata
    with open(f"{dataset_path}.metadata.json", 'r') as f:
        metadata = json.load(f)
    
    # Verify TEMPUS compliance
    if not metadata.get('spatial_sampling', False):
        raise ValueError("❌ Dataset not spatially sampled - not TEMPUS compliant")
    
    if not metadata.get('blackout_enforced', False):
        raise ValueError("❌ Blackout not enforced - data leakage risk")
    
    print(f"✓ Loaded TEMPUS dataset:")
    print(f"  - {len(dataset)} samples")
    print(f"  - Spatial sampling: {metadata['spatial_sampling']}")
    print(f"  - Domain: {metadata.get('domain', 'unknown')}")
    
    return dataset, metadata

## PHASES 2-5: TEMPUS LOOP

**Main training loop:**

```python
def tempus_training_loop(dataset, framework, ap_state):
    """
    Execute TEMPUS protocol for all T₁ samples
    """
    
    results = []
    n_total = len(dataset)
    
    print(f"\n{'='*60}")
    print(f"TEMPUS TRAINING LOOP — {n_total} samples")
    print(f"{'='*60}\n")
    
    for idx, datapoint in enumerate(dataset):
        print(f"\n--- Sample {idx+1}/{n_total} ---")
        
        # Phase 2: Positionnement Rétroactif
        pr_state = phase_pr(datapoint, ap_state)
        
        # Phase 3: Simulation Aveugle
        sa_state = phase_sa(pr_state, framework)
        
        # Phase 4: Validation Empirique
        ve_state = phase_ve(sa_state, datapoint)
        
        # Phase 5: Extraction Causale
        ec_state = phase_ec(ve_state, framework)
        
        # Log result
        result = {
            'T1': datapoint['T1'],
            'prediction': sa_state['prediction'],
            'ground_truth': ve_state['ground_truth'],
            'gap': ve_state['gap'],
            'patterns_identified': ec_state['patterns'],
            'calibration_applied': ec_state['calibration']
        }
        
        results.append(result)
    
    return results
```

### Phase 2: Positionnement Rétroactif

```python
def phase_pr(datapoint, ap_state):
    """
    Phase 2: Positionnement Rétroactif
    
    Jump to T₁ in history, load context [T₁-Δ, T₁]
    """
    
    T1 = datetime.fromisoformat(datapoint['T1'])
    context = datapoint['context_pre']
    
    pr_state = {
        'T1': T1,
        'T0': ap_state['T0'],
        'context': context,
        'blackout_active': True  # No data > T₁ accessible
    }
    
    print(f"T₁: {T1.isoformat()}")
    print(f"Context: {len(context)} datapoints loaded")
    print(f"Blackout: ENFORCED (no future data)")
    
    return pr_state
```

### Phase 3: Simulation Aveugle

```python
def phase_sa(pr_state, framework):
    """
    Phase 3: Simulation Aveugle
    
    Model predicts from T₁ as if it's "now"
    CRITICAL: Model has ZERO knowledge of data > T₁
    """
    
    # Verify blackout
    if not pr_state['blackout_active']:
        raise ValueError("❌ Blackout not active - future leakage risk")
    
    # Generate prediction using only context ≤ T₁
    prediction = framework.predict(
        context=pr_state['context'],
        T_current=pr_state['T1']  # Model thinks T₁ is "now"
    )
    
    sa_state = {
        **pr_state,
        'prediction': prediction,
        'prediction_timestamp': datetime.now()  # When prediction was made (T₀ perspective)
    }
    
    print(f"Prediction: {prediction}")
    
    return sa_state
```

### Phase 4: Validation Empirique

```python
def phase_ve(sa_state, datapoint):
    """
    Phase 4: Validation Empirique
    
    Compare prediction vs reality
    """
    
    prediction = sa_state['prediction']
    ground_truth = datapoint['ground_truth_post']
    
    # Compute gap (domain-specific)
    gap = compute_gap(prediction, ground_truth)
    
    ve_state = {
        **sa_state,
        'ground_truth': ground_truth,
        'gap': gap,
        'gap_components': decompose_gap(gap, prediction, ground_truth)
    }
    
    print(f"Ground truth: {ground_truth}")
    print(f"Gap: {gap:.4f}")
    
    return ve_state

def compute_gap(prediction, ground_truth):
    """
    Domain-specific gap measurement
    """
    
    # Hockey: 0 if exact score, 0.5 if right winner, 1 if wrong
    if 'score' in prediction and 'score' in ground_truth:
        if prediction['score'] == ground_truth['score']:
            return 0.0
        elif prediction['winner'] == ground_truth['winner']:
            return 0.5
        else:
            return 1.0
    
    # Finance: MSE
    elif 'price' in prediction:
        return (prediction['price'] - ground_truth['price']) ** 2
    
    # Climate: temperature delta
    elif 'temperature' in prediction:
        return abs(prediction['temperature'] - ground_truth['temperature'])
    
    # Generic: L1 distance
    else:
        return abs(prediction - ground_truth)

def decompose_gap(gap, prediction, ground_truth):
    """
    Identify which components contributed to gap
    """
    
    components = {}
    
    for key in prediction.keys():
        if key in ground_truth:
            components[key] = abs(prediction[key] - ground_truth.get(key, 0))
    
    return components
```

### Phase 5: Extraction Causale

```python
def phase_ec(ve_state, framework):
    """
    Phase 5: Extraction Causale
    
    Learn from gap - what did model miss?
    """
    
    gap = ve_state['gap']
    gap_components = ve_state['gap_components']
    
    # Identify patterns
    if gap < 0.3:  # Low gap = patterns worked
        patterns_validated = identify_active_patterns(ve_state, framework)
        patterns_missing = []
        calibration = 'reinforce_active_patterns'
        
    elif gap > 0.7:  # High gap = patterns missing
        patterns_validated = []
        patterns_missing = identify_missing_patterns(ve_state, framework)
        calibration = 'add_missing_patterns'
        
    else:  # Medium gap = partial understanding
        patterns_validated = identify_active_patterns(ve_state, framework)
        patterns_missing = identify_missing_patterns(ve_state, framework)
        calibration = 'adjust_weights'
    
    ec_state = {
        **ve_state,
        'patterns': {
            'validated': patterns_validated,
            'missing': patterns_missing
        },
        'calibration': calibration
    }
    
    # Apply calibration
    framework = apply_calibration(framework, ec_state)
    
    print(f"Patterns validated: {len(patterns_validated)}")
    print(f"Patterns missing: {len(patterns_missing)}")
    print(f"Calibration: {calibration}")
    
    return ec_state

def identify_active_patterns(ve_state, framework):
    """
    Which patterns/filters were active in prediction?
    """
    
    # Domain-specific pattern extraction
    # Example for hockey:
    if hasattr(framework, 'filters'):
        active = []
        for filter_name, filter_obj in framework['filters'].items():
            if filter_obj.was_active(ve_state['context']):
                active.append({
                    'name': filter_name,
                    'weight': filter_obj.weight,
                    'contribution': filter_obj.contribution
                })
        return active
    
    return []

def identify_missing_patterns(ve_state, framework):
    """
    What patterns would explain the gap?
    """
    
    gap_components = ve_state['gap_components']
    missing = []
    
    # Analyze which variables had largest error
    sorted_components = sorted(
        gap_components.items(), 
        key=lambda x: x[1], 
        reverse=True
    )
    
    # Top 3 error sources
    for var, error in sorted_components[:3]:
        missing.append({
            'variable': var,
            'error': error,
            'hypothesis': generate_hypothesis(var, error, ve_state)
        })
    
    return missing

def apply_calibration(framework, ec_state):
    """
    Update framework based on EC analysis
    """
    
    calibration = ec_state['calibration']
    
    if calibration == 'reinforce_active_patterns':
        # Increase weight of validated patterns
        for pattern in ec_state['patterns']['validated']:
            framework.adjust_weight(pattern['name'], delta=+0.1)
    
    elif calibration == 'add_missing_patterns':
        # Add new patterns to framework
        for pattern in ec_state['patterns']['missing']:
            framework.add_pattern(pattern)
    
    elif calibration == 'adjust_weights':
        # Fine-tune existing weights
        framework.optimize_weights(ec_state)
    
    return framework
```

```python
def analyze_tempus_results(results):
    """
    Analyze TEMPUS training results
    """
    
    analysis = {
        'total_samples': len(results),
        'avg_gap': np.mean([r['gap'] for r in results]),
        'gap_std': np.std([r['gap'] for r in results]),
        'gap_distribution': compute_gap_distribution(results),
        'patterns_discovered': aggregate_patterns(results),
        'temporal_coverage': analyze_temporal_coverage(results)
    }
    
    # Success rate
    low_gap_count = sum(1 for r in results if r['gap'] < 0.3)
    analysis['success_rate'] = low_gap_count / len(results)
    
    print(f"\n{'='*60}")
    print(f"TEMPUS TRAINING ANALYSIS")
    print(f"{'='*60}")
    print(f"Total samples: {analysis['total_samples']}")
    print(f"Average gap: {analysis['avg_gap']:.4f}")
    print(f"Success rate (gap < 0.3): {analysis['success_rate']*100:.1f}%")
    print(f"Patterns discovered: {len(analysis['patterns_discovered'])}")
    
    return analysis

def compute_gap_distribution(results):
    """Gap histogram"""
    gaps = [r['gap'] for r in results]
    
    bins = [0, 0.3, 0.7, 1.0]
    labels = ['Low (0-0.3)', 'Medium (0.3-0.7)', 'High (0.7-1.0)']
    
    distribution = {}
    for i, label in enumerate(labels):
        count = sum(1 for g in gaps if bins[i] <= g < bins[i+1])
        distribution[label] = count
    
    return distribution

def aggregate_patterns(results):
    """Collect all discovered patterns"""
    all_patterns = []
    
    for result in results:
        all_patterns.extend(result['patterns_identified']['validated'])
        all_patterns.extend(result['patterns_identified']['missing'])
    
    # Deduplicate and count
    pattern_counts = {}
    for pattern in all_patterns:
        key = pattern.get('name', str(pattern))
        pattern_counts[key] = pattern_counts.get(key, 0) + 1
    
    # Sort by frequency
    sorted_patterns = sorted(
        pattern_counts.items(), 
        key=lambda x: x[1], 
        reverse=True
    )
    
    return sorted_patterns
```

```python
def validate_causal_understanding(framework, test_set):
    """
    Test if model learned causal patterns (not chronology)
    """
    
    validation_tests = []
    
    # Test 1: Novel temporal positions
    # Use T₁ points NOT in training set
    novel_positions = generate_novel_T1_points(test_set)
    test1_results = [
        phase_sa({'T1': T1, 'context': ctx, 'blackout_active': True}, framework)
        for T1, ctx in novel_positions
    ]
    test1_gap = np.mean([compute_gap(r['prediction'], truth) 
                         for r, truth in zip(test1_results, test_set)])
    
    validation_tests.append({
        'test': 'Novel temporal positions',
        'gap': test1_gap,
        'passed': test1_gap < 0.5
    })
    
    # Test 2: Counterfactual scenarios
    # Modify context variables, see if predictions adapt correctly
    counterfactual_results = test_counterfactuals(framework, test_set)
    validation_tests.append({
        'test': 'Counterfactual adaptation',
        'passed': counterfactual_results['coherent']
    })
    
    # Test 3: Pattern generalization
    # Apply to different domain with same causal structure
    generalization_results = test_generalization(framework, test_set)
    validation_tests.append({
        'test': 'Pattern generalization',
        'passed': generalization_results['transferable']
    })
    
    all_passed = all(t['passed'] for t in validation_tests)
    
    print(f"\n{'='*60}")
    print(f"CAUSAL UNDERSTANDING VALIDATION")
    print(f"{'='*60}")
    for test in validation_tests:
        status = "✓ PASS" if test['passed'] else "✗ FAIL"
        print(f"{status}: {test['test']}")
    
    return all_passed, validation_tests
```

## ANTI-PATTERNS

❌ **Sequential iteration:** Loop T₁ in chronological order violates TEMPUS
❌ **Future leakage:** Context includes data > T₁
❌ **No calibration:** Run loop without updating framework
❌ **Ignore gaps:** Don't analyze what model missed
❌ **Over-calibration:** Update weights after every sample (noise)
